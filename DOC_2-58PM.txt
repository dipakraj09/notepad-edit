/* ── FORMAT ── */
function fmt(bytes){
  if(bytes < 1024) return bytes + ' B';
  if(bytes < 1024*1024) return (bytes/1024).toFixed(1) + ' KB';
  return (bytes/(1024*1024)).toFixed(2) + ' MB';
}

/* ── COMPRESS ── */
async function startCompress(){
  if(!pdfFile) return;
  const quality = parseInt(document.getElementById('qualitySlider').value) / 100;
  const btn = document.getElementById('compressBtn');
  btn.disabled = true;
  document.getElementById('resultCard').classList.remove('show');
  document.getElementById('progressWrap').classList.add('show');
  try {
    const ab = await pdfFile.arrayBuffer();
    const src = await pdfjsLib.getDocument({ data: ab }).promise;
    const total = src.numPages;
    const out = await PDFLib.PDFDocument.create();
    setProgress(5, 'Starting compression…');

    for(let n=1; n<=total; n++){
      setProgress(5 + ((n-1)/total)*85, `Rendering page ${n} of ${total}…`);
      const pg = await src.getPage(n);
      const vp = pg.getViewport({ scale: imageScale * 2.0 });
      const cv = document.createElement('canvas');
      cv.width = vp.width; cv.height = vp.height;
      await pg.render({ canvasContext: cv.getContext('2d'), viewport: vp }).promise;
      const imgB64 = cv.toDataURL('image/jpeg', quality).split(',')[1];
      const imgBytes = Uint8Array.from(atob(imgB64), c => c.charCodeAt(0));
      const jpg = await out.embedJpg(imgBytes);
      const orig = pg.getViewport({ scale: 1.0 });
      const newPg = out.addPage([orig.width, orig.height]);
      newPg.drawImage(jpg, { x:0, y:0, width:orig.width, height:orig.height });
    }

    setProgress(95, 'Saving file…');
    const bytes = await out.save();
    compressedBlob = new Blob([bytes], { type:'application/pdf' });
    setProgress(100, '✓ Done!');

    const ori = pdfFile.size, nw = compressedBlob.size;
    const red = Math.round((1 - nw/ori)*100);
    document.getElementById('origSize').textContent = fmt(ori);
    document.getElementById('newSize').textContent  = fmt(nw);
    document.getElementById('reduction').textContent = red > 0 ? red + '%' : '~0%';

    const url = URL.createObjectURL(compressedBlob);
    document.getElementById('downloadBtn').onclick = () => {
      const a = document.createElement('a');
      a.href = url;
      a.download = pdfFile.name.replace('.pdf','_compressed.pdf');
      a.click();
    };
    setTimeout(() => {
      document.getElementById('progressWrap').classList.remove('show');
      document.getElementById('resultCard').classList.add('show');
    }, 500);
  } catch(err){
    console.error(err);
    alert('Error: ' + err.message);
    document.getElementById('progressWrap').classList.remove('show');
  }
  btn.disabled = false;
}

function setProgress(pct, status){
  document.getElementById('progressBar').style.width = pct + '%';
  document.getElementById('progressPct').textContent = Math.round(pct) + '%';
  document.getElementById('progressStatus').textContent = status;
  document.getElementById('progressText').textContent = pct < 100 ? 'Compressing pages…' : '✓ Complete';
}
